518 template< class t>
519 int CheckAvailable(std::vector<t *> res,double startingHour, double endHour,int day, int month)
520 {
521  for(size_t i =0; i < res.size(); i++)
522  {
523  if(res.at(i)->getDay() == day && res.at(i)->getMonth() == month)
524  {
525  //res- values from reservation already made
526  double resStart = res.at(i)->getStartingHour();
527  double resEnd = calculateEndHour(resStart,res.at(i)->getDuration());
528 
529  if(startingHour >= resStart && endHour <= resEnd)
530  // tempo da reserva esta em espaço ocupado
531  throw(InsideRes(resStart,endHour));
532  else if(startingHour <= resStart && endHour >= resStart)
533  //tempo da reserva entra em espaço ocupado
534  throw(EndHourInsideRes(resStart,resEnd));
535  else if(startingHour >= resStart && endHour >= resEnd && startingHour <= resEnd)
536  //tempo da reserva esta a meio
537  throw(StartHourInsideRes(startingHour, endHour));
538  }
539  }
540  return 0;
