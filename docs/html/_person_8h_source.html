557 template< class t>
558 int CheckAvailable(std::vector<t *> res,double startingHour, double endHour,int day, int month)
559 {
560  for(size_t i =0; i < res.size(); i++)
561  {
562  if(res.at(i)->getDay() == day && res.at(i)->getMonth() == month)
563  {
564  //res- values from reservation already made
565  double resStart = res.at(i)->getStartingHour();
566  double resEnd = calculateEndHour(resStart,res.at(i)->getDuration());
567 
568  if(startingHour >= resStart && endHour <= resEnd)
569  // tempo da reserva esta em espaço ocupado
570  throw(InsideRes(resStart,endHour));
571  else if(startingHour <= resStart && endHour >= resStart)
572  //tempo da reserva entra em espaço ocupado
573  throw(EndHourInsideRes(resStart,resEnd));
574  else if(startingHour >= resStart && endHour >= resEnd && startingHour <= resEnd)
575  //tempo da reserva esta a meio
576  throw(StartHourInsideRes(startingHour, endHour));
577  }
578  }
579  return 0;
580 }
