535 int CheckAvailable(std::vector<t *> res,double startingHour, double endHour,int day, int month)
536 {
537  for(size_t i =0; i < res.size(); i++)
538  {
539  if(res.at(i)->getDay() == day && res.at(i)->getMonth() == month)
540  {
541  //res- values from reservation already made
542  double resStart = res.at(i)->getStartingHour();
543  double resEnd = calculateEndHour(resStart,res.at(i)->getDuration());
544 
545  if(startingHour >= resStart && endHour <= resEnd)
546  // tempo da reserva esta em espaço ocupado
547  throw(InsideRes(resStart,endHour));
548  else if(startingHour <= resStart && endHour >= resStart)
549  //tempo da reserva entra em espaço ocupado
550  throw(EndHourInsideRes(resStart,resEnd));
551  else if(startingHour >= resStart && endHour >= resEnd && startingHour <= resEnd)
552  //tempo da reserva esta a meio
553  throw(StartHourInsideRes(startingHour, endHour));
554  }
555  }
556  return 0;
